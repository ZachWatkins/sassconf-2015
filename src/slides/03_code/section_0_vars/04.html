A Slightly Optimized Mixin

@mixin fpo(
  $size: 100%, $bg-color: slategray, $color: #fff, $text: 'FPO'
) {
  background-color: $bg-color;
  position: relative;

  &:before {
    content: '';
    display: block;
    padding-bottom: $size;
  }
  
  &:after {
    content: $text;
    color: $color;
    font: 20px/1 Helvetica;
    position: absolute;
    top: 50%;
    left: 50%;
    transform: translate(-50%, -50%);
  }
}

@include fpo(50%, tomato);
